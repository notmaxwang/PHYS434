#Imported all the libraries required for generating the random numbers, plotting the graphs, and displaying the statistics.
import numpy as np
import matplotlib
import matplotlib.pyplot as plt
import scipy
from scipy import stats

#Sets up the size of the olot graphs to a reasonable size.
plt.rcParams["figure.figsize"] = (15,10)

#This sets up a Gaussian distribution with 100,000 points using the stats.norm package from the scipy library that we imported.
#Then, a histogram of these data points are plotted using the pyplot function, which is called as plt. 
d = stats.norm.rvs(loc = 3.0, scale = .01, size = 100000)
fig, ax = plt.subplots(1, 1)
ax.hist(d,50, density=True)
plt.tick_params(labelsize = 24)
plt.xlim([2.95,3.05])
x = np.linspace(2.95,3.05,1000)
ax.plot(x,stats.norm.pdf(x,loc = 3., scale = 0.01),linewidth = 8,alpha = 0.7)
plt.show()
